# Typescript 유틸리티 타입

```typescript
// keyof

interface User {
  id: number;
  name: string;
  age: number;
  gender: 'm' | 'f';
}

type UserKey = keyof User; // 'id' | 'name' | 'age' | 'gender';

const myKey: UserKey = 'name';


```

```typescript
// Partial<T> 
// 프로퍼티를 모두 옵셔널로 바꿔줌

interface User {
  id: number;
  name: string;
  age: number;
  gender: 'm' | 'f';
}

let admin: Partial<User> = {
  id: 1,
  name: 'Bob',
}

```

<br />

```typescript
// Required<T>
// 모든 프로퍼티를 필수로 바꿔줌

interface User {
  id: number;
  name: string;
  age?: number;
}

let admin: Required<User> = {
  id: 1,
  name: 'Bob',
  age: 1 // 옵셔널이지만 필수로 요구됨
}

```

// Readonly<T>

```typescript
// Record<K,T>

type Grade = '1' | '2' | '3' ;
type Scroe = 'A' | 'B' | 'C' ;


const score: Record<Grade, Score> = {
  1: 'A',
  2: 'B',
  3: 'C',
}

interface User {
  id: number;
  name: string;
  age: number;
}

function isValid(user: User) {
  const result: Record<keyof User, boolean> = {
    id: user.id > 0,
    name: user.name !== '',
    age: user.age > 0
  };
  
  return result;
}

```

```typescript
// Pick<T,K>
// Omit<T,K>
// Exclude<T1, T2>  t1의 타입중에 t2타입과 겹치는걸 제거

interface User {
  id: number;
  name: string;
  age: number;
  gender: 'M' | 'F';
}

// pick 유저에서 id와 name만 가져온다
const admin: Pick<User, "id" | "name"> = {
  id: 0,
  name: 'han',
}


// omit 유저에서 age와 gender만 제외하고 온다
const admin:Omit<User, "age" | "gender"> = {
  id: 0,
  name: 'han',
}

type T1 = string | number | boolean;
type T2 = Exclude<T1, string | number>; // string과 number제외 boolean만 남음

```

// NonNullable<Type> // null과 undefined를 제외한 타입생성

type T1 = string | null | undefined | void;
type T2 = NonNullable<T1>; // string | void만 남게됨
