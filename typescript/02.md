# Typescript 인터페이스

```typescript
// 프로퍼티를 정의해서 객체로 표현할때

type Score = "1등급" | "2등급" | "3등급";

interface User {
  name: string;
  age: number;
  gender?: string; // optional
  readonly birth: string; // 수정불가
  [grade: number]: Score; // 다중 프로버티 지정
}

let user: User = {
  name: "han",
  age: 29,
  birth: 1994,
  1: "1등급",
  2: "4등급", // err: Score에서 지정해준값 강제
};

user.age; // 29
user.birth = 2000; // err
```

```typescript
// 인터페이스로 함수표현

interface Add {
  (num1: number, num2: number): number;
}

const add: Add = function (num1, num2) {
  return num1 + num2;
};

add(1, 2);
add(1, "오징어", 3); // err

interface IsAdult {
  (age: number): boolean;
}

const isAdult: IsAdult = (age) => {
  return age > 19;
};

isAdult(29); // true
```

```typescript
// 인터페이스로 클래스 정의
// implements

interface Car {
  color: string;
  wheels: number;
  start(): void;
}

// 여러개도 extends 가능
interface UpgradeCar extends Car, Bike {
  door: number;
}

class KIA implements UpgradeCar {
  color;
  wheels: 4;
  door: 4,
  start() {
    console.log("시동완료");
  }

  constructor(choiceColor: string) {
    this.color = choiceColor;
  }
}

const myCar = new KIA("blue");

myCar; // { "wheels": 4, color: 'blue', door: 4 }
myCar.start(); // "시동완료"
```
