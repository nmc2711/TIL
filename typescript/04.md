# Typescript 리터럴, 유니온/교차 타입

```typescript
// Literal Types
const fixedUserName = "hwang"; // 변할수없는 const는 "hwang"타입으로 고정
let userName: string | number = "hwang"; // 하지만 변할수있는 let은 "hwang"타입이 아닌 string | number 타입 유동
```

<br />

```typescript
// Union Types | or
// 동일한 속성의 타입을 다르게하여 만드는 식별가능한 유니온 타입

interface Car {
  name: "car";
  color: string;
  start(): void;
}

interface Mobile {
  name: "mobile";
  color: string;
  call(): void;
}

function getGift(gift: Car | Mobile) {
  console.log(gift.color); // 공통 속성 ok

  // but start() call()은 공통되지않음 그러므로 동일한 속성의 다른타입인 name으로 구분
  if (gift.name === "car") {
    gift.start();
  } else {
    gift.call();
  }
}
```

```typescript
// Intersection Types & and

interface Car {
  name: string;
  start(): void;
}

interface Toy {
  name: string;
  color: string;
  price: number;
}

const toyCar: Toy & Car = {
  name: "아반뗴",
  color: "red",
  price: 2000,
  start() {
    console.log("부릉..");
  },
};
```
